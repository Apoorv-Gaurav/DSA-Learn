class Solution {
    public int numberOfSpecialChars(String word) {
        int[] hash = new int[123];
        for(int i=0;i<word.length();i++){
            int ch = word.charAt(i);
            hash[ch] = 1;
        }
        int c = 0;
        for(int i=65;i<=90;i++){
            if(hash[i]==1 && hash[i+32]==1){
                c++;
            }
        }
        return c;
    }
}
