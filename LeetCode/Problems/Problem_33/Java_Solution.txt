class Solution {
    public int search(int[] nums, int target) {

        int left = 0;
        int right = nums.length - 1;

        int last = nums[right];

        while(left <= right){

            int mid = left + (right - left) / 2;

            if(nums[mid] == target){
                return mid;
            }

            // target and mid both in left side
            if(target > last && nums[mid] > last){

                if(target > nums[mid]){
                    left = mid + 1;
                }
                else{
                    right = mid - 1;
                }
            }

            // target in left side, mid in right side
            else if(target > last && nums[mid] <= last){
                right = mid - 1;
            }

            // target in right side, mid in left side
            else if(target <= last && nums[mid] > last){
                left = mid + 1;
            }

            // target and mid both in right side
            else{

                if(target > nums[mid]){
                    left = mid + 1;
                }
                else{
                    right = mid - 1;
                }
            }
        }

        return -1;
    }
}
